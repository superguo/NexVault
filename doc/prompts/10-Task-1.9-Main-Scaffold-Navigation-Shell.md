 * IMPORTANT: Check the actual NexVaultBottomNavBar API from core-ui.
 *
 * Possible API patterns for NexVaultBottomNavBar:
 *
 * Pattern A — It's a fully self-contained composable that takes items:
 * ```
 * NexVaultBottomNavBar(
 *     items = listOf("Home", "History", "DApp", "NFT", "Settings"),
 *     selectedIndex = selectedTabIndex,

 * ```
 * NexVaultBottomNavBar {
 *     tabs.forEach { tab ->
 *         NavigationBarItem(...)

 * ```

 * ```

 * NavigationBar + NavigationBarItem composables.

@Composable
fun MainScreen(
    modifier: Modifier = Modifier,
) {
    val tabNavController = rememberNavController()
    val navBackStackEntry by tabNavController.currentBackStackEntryAsState()
    val currentDestination = navBackStackEntry?.destination
    val tabs = MainTab.entries // or MainTab.values() if using older Kotlin

    Scaffold(
        modifier = modifier,
        bottomBar = {
            // OPTION 1: Use NexVaultBottomNavBar from core-ui if it fits
            // OPTION 2: Build directly with NavigationBar (shown below as fallback)
            //
            // The AI should TRY to use NexVaultBottomNavBar first.
            // If it doesn't fit, use NavigationBar directly.

            NavigationBar(
                containerColor = MaterialTheme.colorScheme.surface,
                contentColor = MaterialTheme.colorScheme.onSurface,
            ) {
                tabs.forEach { tab ->
                    val isSelected = currentDestination?.hierarchy?.any {
                        it.route == tab.route
                    } == true

                    NavigationBarItem(
                        selected = isSelected,
                        onClick = {
                            tabNavController.navigate(tab.route) {
                                // Pop up to the start destination of the graph to
                                // avoid building up a large stack of destinations
                                popUpTo(tabNavController.graph.findStartDestination().id) {
                                    saveState = true
                                }
                                // Avoid multiple copies of the same destination
                                launchSingleTop = true
                                // Restore state when re-selecting a previously selected item
                                restoreState = true
                            }
                        },
                        icon = {
                            Icon(
                                imageVector = if (isSelected) tab.selectedIcon else tab.unselectedIcon,
                                contentDescription = tab.contentDescription,
                            )
                        },
                        label = {
                            Text(
                                text = tab.label,
                                style = MaterialTheme.typography.labelSmall,
                            )
                        },
                        colors = NavigationBarItemDefaults.colors(
                            selectedIconColor = MaterialTheme.colorScheme.primary,
                            selectedTextColor = MaterialTheme.colorScheme.primary,
                            unselectedIconColor = MaterialTheme.colorScheme.onSurfaceVariant,
                            unselectedTextColor = MaterialTheme.colorScheme.onSurfaceVariant,
                            indicatorColor = MaterialTheme.colorScheme.primaryContainer,
                        ),
                    )
                }
            }
        },
    ) { innerPadding ->
        // Nested NavHost for tab content
        NavHost(
            navController = tabNavController,
            startDestination = MainTab.HOME.route,
            modifier = Modifier.padding(innerPadding),

### 9. Common Issues to Watch For:

**Issue: NexVaultBottomNavBar has hardcoded tabs**
If the core-ui `NexVaultBottomNavBar` was generated with hardcoded 5 tabs and no
callback mechanism, you have two options:
- Refactor it to accept parameters (preferred, but might break other things)
- Ignore it and use `NavigationBar` + `NavigationBarItem` directly in `MainScreen`

The second option is simpler and avoids breaking existing code. Use it if the
existing component doesn't fit.

**Issue: Icons.Outlined not found**
If `Icons.Outlined` types don't resolve, the `material-icons-extended` dependency
is missing. Either add it or use `Icons.Filled` for both states:
```kotlin
HOME(
    selectedIcon = Icons.Filled.Home,
    unselectedIcon = Icons.Filled.Home, // same icon, differentiated by tint
)
```
The `NavigationBarItem` already handles tint differentiation between selected/unselected.

**Issue: Two NavHosts conflict**
The root `NavHost` in `NexVaultApp` and the tab `NavHost` in `MainScreen` are
completely independent. They use separate `NavController` instances. This is the
correct pattern for bottom navigation inside a broader app navigation structure.
Make sure `MainScreen` creates its own `rememberNavController()` and does NOT
receive the root NavController.

**Issue: Bottom bar hidden when navigating to detail screens**
This is a Phase 2 concern. When a tab navigates to a detail screen (e.g.,
HomeScreen → TokenDetailScreen), the bottom bar should still be visible since the
detail screen is within the tab's nested graph. This is handled automatically because
the detail screen will be inside the tab's `NavHost`, not the root `NavHost`.
For nested detail screens where you DON'T want the bottom bar (e.g., Send flow),
those will be navigated via the root NavController (navigating outside of MainScreen).
This will be addressed in Phase 2.

**Issue: Tab content flickers on selection**
If tab content flickers when switching, ensure `restoreState = true` is set in the
`navigate()` call and `saveState = true` is set in the `popUpTo` block.

**Issue: First tab not selected on launch**
The `startDestination = MainTab.HOME.route` in the tab NavHost ensures the Home tab
is shown by default. The `NavigationBarItem.selected` check uses `currentDestination?.hierarchy`
which correctly highlights the Home tab on first render.

**Issue: Back press on non-Home tab should go to Home tab**
The `popUpTo(tabNavController.graph.findStartDestination().id)` in the navigate call
handles this. When on a non-Home tab, pressing system back will pop to the Home tab
(the start destination of the tab NavHost) rather than closing the app. On the Home tab,
system back will exit the app (or go to the unlock screen, depending on the back stack).
