/**
 * Displays mnemonic words in a 4×3 grid (4 rows, 3 columns).
 * Each cell shows the word number and the word.
 */
@Composable
fun MnemonicGrid(
    words: List<String>,
    modifier: Modifier = Modifier,
) {
    LazyVerticalGrid(
        columns = GridCells.Fixed(3),
        modifier = modifier,
        horizontalArrangement = Arrangement.spacedBy(8.dp),
        verticalArrangement = Arrangement.spacedBy(8.dp),
        userScrollEnabled = false, // Grid should not scroll — all 12 words visible at once
    ) {
        itemsIndexed(words) { index, word ->
            MnemonicWordCell(
                number = index + 1,
                word = word,
            )
        }
    }
}

/**
 * ViewModel for the Verify Mnemonic screen.
 *
 * Receives the walletId via SavedStateHandle (navigation argument).
 * Loads the mnemonic from the repository, shuffles a copy for verification.
 * User taps words in the correct order. If all words are selected correctly,
 * the Confirm button becomes enabled.
 *
 * On incorrect selection, shows an error state with shake animation trigger,
 * then resets after a short delay.
 */
@HiltViewModel
class VerifyMnemonicViewModel @Inject constructor(
    savedStateHandle: SavedStateHandle,
    private val walletRepository: WalletRepository,
    // ^ Adjust type — import the actual WalletRepository interface from domain
) : ViewModel() {

    data class UiState(
        val originalWords: List<String> = emptyList(),
        val shuffledWords: List<String> = emptyList(),
        val selectedWords: List<String> = emptyList(),
        val availableWords: List<String> = emptyList(), // shuffled minus selected
        val isVerified: Boolean = false,
        val isError: Boolean = false,
        val isLoading: Boolean = true,
        val errorMessage: String? = null,
    )

    sealed interface NavigationEvent {
        data object NavigateToSetPin : NavigationEvent
    }

    private val walletId: String = savedStateHandle.get<String>("walletId") ?: ""
    // ^ Adjust the key name to match the navigation argument name

    private val _uiState = MutableStateFlow(UiState())
    val uiState: StateFlow<UiState> = _uiState.asStateFlow()

    private val _navigationEvent = MutableSharedFlow<NavigationEvent>(extraBufferCapacity = 1)
    val navigationEvent: SharedFlow<NavigationEvent> = _navigationEvent.asSharedFlow()

    init {
        loadMnemonic()
    }

    private fun loadMnemonic() {
        viewModelScope.launch {
            _uiState.update { it.copy(isLoading = true) }

            // Get mnemonic for backup/verification
            val result = walletRepository.getMnemonicForBackup(walletId)
            // ^ Adjust method name to match actual WalletRepository API

            when (result) {
                is DataResult.Success -> {
                    val words = result.data
                    // ^ Adjust field name
                    val shuffled = words.shuffled()
                    _uiState.update {
                        it.copy(
                            originalWords = words,
                            shuffledWords = shuffled,
                            availableWords = shuffled,
                            selectedWords = emptyList(),
                            isLoading = false,
                        )
                    }
                }
                is DataResult.Error -> {
                    _uiState.update {
                        it.copy(
                            isLoading = false,
                            errorMessage = "Failed to load mnemonic",
                        )
                    }
                }
                else -> {}
            }
        }
    }

    /**
     * Called when the user taps a word from the available (shuffled) words pool.
     *
     * Checks if the tapped word matches the next expected word in the original
     * mnemonic sequence. If correct, adds it to the selected list. If all 12
     * are selected correctly, marks as verified.
     *
     * If incorrect, triggers error state and resets after a delay.
     */
    fun onWordSelected(word: String) {
        val currentState = _uiState.value
        if (currentState.isVerified || currentState.isError) return

        val nextExpectedIndex = currentState.selectedWords.size
        val expectedWord = currentState.originalWords.getOrNull(nextExpectedIndex) ?: return

        if (word == expectedWord) {
            // Correct word selected
            val newSelected = currentState.selectedWords + word
            val newAvailable = currentState.availableWords.toMutableList().apply {
                remove(word)  // Remove first occurrence only
            }

            val isComplete = newSelected.size == currentState.originalWords.size

            _uiState.update {
                it.copy(
                    selectedWords = newSelected,
                    availableWords = newAvailable,
                    isVerified = isComplete,
                    isError = false,
                )
            }
        } else {
            // Wrong word — trigger error and reset
            _uiState.update { it.copy(isError = true) }

            viewModelScope.launch {
                delay(1500) // Show error state for 1.5 seconds
                resetVerification()
            }
        }
    }

    /**
     * Called when the user taps a word from the selected words to deselect it.
     * Only the last selected word can be removed (to maintain order).
     */
    fun onSelectedWordRemoved(word: String) {
        val currentState = _uiState.value
        if (currentState.isVerified || currentState.isError) return

        // Only allow removing the last word
        if (currentState.selectedWords.lastOrNull() != word) return

        val newSelected = currentState.selectedWords.dropLast(1)
        val newAvailable = currentState.availableWords + word

        _uiState.update {
            it.copy(
                selectedWords = newSelected,
                availableWords = newAvailable,
                isVerified = false,
            )
        }
    }

    private fun resetVerification() {
        val currentState = _uiState.value
        _uiState.update {
            it.copy(
                selectedWords = emptyList(),
                availableWords = currentState.shuffledWords,
                isVerified = false,
                isError = false,
            )
        }
    }

    fun onConfirmClicked() {
        if (_uiState.value.isVerified) {
            _navigationEvent.tryEmit(NavigationEvent.NavigateToSetPin)
        }
    }

    fun onResetClicked() {
        resetVerification()
    }
}

## Key Rules:

1. **Inspect actual core-ui components** — use real composable names and parameter signatures
2. **Inspect actual domain use cases** — use real invocation patterns (operator invoke, execute, etc.)
3. **Inspect actual domain models** — use real field names (DataResult.Success.data, etc.)
4. **No hardcoded colors** — all colors come from MaterialTheme.colorScheme
5. **No hardcoded strings** — ideally use string resources, but for Phase 1 inline strings
   are acceptable. Move to resources in Phase 3.
6. **FLAG_SECURE on CreateWalletScreen** — prevent screenshots of mnemonic
7. **All composables have @Preview functions** — at least one preview per screen
8. **ViewModel uses StateFlow for state, SharedFlow for one-time events** (navigation)
9. **collectAsStateWithLifecycle** — use lifecycle-aware collection, not plain collectAsState
10. **Separate content composables** — each screen has a `XxxScreen` (with ViewModel) and

## Common Issues to Watch For:

**Issue: weight() inside verticalScroll()**
Modifier.weight() does not work inside a scrollable container. Fix by either:
(a) removing verticalScroll and ensuring content fits, or (b) placing the bottom
button in Scaffold's bottomBar slot, or (c) wrapping non-scrollable content outside
the scrollable area. **Option (b) is recommended** for CreateWallet and VerifyMnemonic.

**Issue: FlowRow import**
Use `androidx.compose.foundation.layout.FlowRow` with `@OptIn(ExperimentalLayoutApi::class)`.
Do NOT use `com.google.accompanist.flowlayout.FlowRow` (deprecated). If FlowRow is not
available, use a `LazyVerticalGrid(columns = GridCells.Adaptive(80.dp))` for available words.

**Issue: NexVaultButton missing enabled parameter**
If the core-ui `NexVaultButton` doesn't have an `enabled` param, add it:
```kotlin
@Composable
fun NexVaultButton(
    text: String,  // or content lambda
    onClick: () -> Unit,
    modifier: Modifier = Modifier,
    enabled: Boolean = true,  // ADD THIS
) { ... }
```
Apply the same to `NexVaultOutlinedButton`.

**Issue: SavedStateHandle key for walletId**
The key used in `savedStateHandle.get<String>("walletId")` must match the navigation
argument name in the route definition. If the route is
`"onboarding/verify_mnemonic/{walletId}"`, the key is `"walletId"`.

**Issue: LocalView and FLAG_SECURE**
The `LocalView` approach for FLAG_SECURE works when the composable is hosted in an Activity.
Import `androidx.compose.ui.platform.LocalView`. The null-safe cast `as?` handles cases
where the context is not an Activity.
