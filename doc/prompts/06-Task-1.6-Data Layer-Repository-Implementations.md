// In WalletRepositoryImpl
import org.web3j.crypto.Keys
import org.web3j.crypto.ECKeyPair
import java.math.BigInteger

private fun deriveAddressFromPrivateKey(privateKeyHex: String): String {
    val privateKeyBigInt = BigInteger(privateKeyHex, 16)

    return "0x${Keys.getAddress(keyPair)}"

**Issue: DataStore flows vs suspend functions**
