/**
 * ViewModel for the Unlock screen.
 *
 * Handles PIN verification, biometric authentication trigger,
 * and lockout after 5 consecutive failed attempts.
 *
 * Lockout behavior (ref: AC-1.5):
 * - After 5 consecutive wrong PIN attempts, lock the user out for 30 seconds.
 * - Display a countdown timer during lockout.
 * - After the countdown, allow PIN entry again (attempt counter resets).
 *
 * Biometric behavior:
 * - If biometric is enabled in preferences, auto-trigger biometric prompt on screen load.
 * - Provide a "Use Biometric" button as a fallback to re-trigger.
 * - Successful biometric auth navigates to main graph directly (no PIN needed).
 *
 * IMPORTANT: Adjust all use case and repository calls to match actual signatures.
 */
@HiltViewModel
class UnlockViewModel @Inject constructor(
    private val verifyPinUseCase: VerifyPinUseCase,
    private val authRepository: AuthRepository,
    private val preferences: NexVaultPreferences,
    // ^ Adjust types to match actual class names
) : ViewModel() {

    companion object {
        private const val MAX_ATTEMPTS = 5
        private const val LOCKOUT_DURATION_SECONDS = 30
        private const val PIN_LENGTH = 6
    }

    data class UiState(
        val pin: String = "",
        val error: String? = null,
        val isLockedOut: Boolean = false,
        val lockoutRemainingSeconds: Int = 0,
        val isBiometricAvailable: Boolean = false,
        val isBiometricEnabled: Boolean = false,
        val isVerifying: Boolean = false,
        val isShakeError: Boolean = false,
        val failedAttempts: Int = 0,
    )

    sealed interface NavigationEvent {
        data object NavigateToMain : NavigationEvent
    }

    sealed interface BiometricEvent {
        data object ShowBiometricPrompt : BiometricEvent
    }

    private val _uiState = MutableStateFlow(UiState())
    val uiState: StateFlow<UiState> = _uiState.asStateFlow()

    private val _navigationEvent = MutableSharedFlow<NavigationEvent>(extraBufferCapacity = 1)
    val navigationEvent: SharedFlow<NavigationEvent> = _navigationEvent.asSharedFlow()

    private val _biometricEvent = MutableSharedFlow<BiometricEvent>(extraBufferCapacity = 1)
    val biometricEvent: SharedFlow<BiometricEvent> = _biometricEvent.asSharedFlow()

    private var lockoutJob: Job? = null

    init {
        checkBiometricStatus()
    }

    private fun checkBiometricStatus() {
        viewModelScope.launch {
            val available = authRepository.isBiometricAvailable()
            // ^ Adjust method name to match actual API

            val enabled = preferences.isBiometricEnabled.first()
            // ^ Adjust: might be a suspend function, a Flow, or a property
            // If it's a Flow, use .first() to get current value

            _uiState.update {
                it.copy(
                    isBiometricAvailable = available,
                    isBiometricEnabled = available && enabled,
                )
            }

            // Auto-trigger biometric prompt if enabled
            if (available && enabled) {
                _biometricEvent.tryEmit(BiometricEvent.ShowBiometricPrompt)
            }
        }
    }

    /**
     * Called when a digit is pressed on the PIN keypad.
     * When PIN reaches 6 digits, automatically verifies.
     */
    fun onDigitPressed(digit: Int) {
        val currentState = _uiState.value
        if (currentState.isLockedOut || currentState.isVerifying || currentState.pin.length >= PIN_LENGTH) return

        val newPin = currentState.pin + digit.toString()
        _uiState.update { it.copy(pin = newPin, error = null, isShakeError = false) }

        if (newPin.length == PIN_LENGTH) {
            verifyPin(newPin)
        }
    }

    /**
     * Called when the backspace/delete button is pressed.
     */
    fun onBackspacePressed() {
        val currentState = _uiState.value
        if (currentState.isLockedOut || currentState.isVerifying || currentState.pin.isEmpty()) return

        _uiState.update {
            it.copy(
                pin = it.pin.dropLast(1),
                error = null,
                isShakeError = false,
            )
        }
    }

    private fun verifyPin(pin: String) {
        viewModelScope.launch {
            _uiState.update { it.copy(isVerifying = true) }

            try {
                val isValid = verifyPinUseCase(pin)
                // ^ Adjust: might be verifyPinUseCase.invoke(pin),
                // might return Result<Boolean>, Boolean, or Unit (throws on failure)
                // If it returns Result, unwrap appropriately.

                if (isValid) {
                    // PIN correct — reset attempts and navigate
                    _uiState.update {
                        it.copy(
                            isVerifying = false,
                            failedAttempts = 0,
                            error = null,
                        )
                    }
                    _navigationEvent.tryEmit(NavigationEvent.NavigateToMain)
                } else {
                    handleFailedAttempt()
                }
            } catch (e: Exception) {
                // If verifyPinUseCase throws on invalid PIN rather than returning false
                handleFailedAttempt()
            }
        }
    }

    private fun handleFailedAttempt() {
        val newFailedAttempts = _uiState.value.failedAttempts + 1

        if (newFailedAttempts >= MAX_ATTEMPTS) {
            // Trigger lockout
            _uiState.update {
                it.copy(
                    isVerifying = false,
                    pin = "",
                    failedAttempts = newFailedAttempts,
                    isLockedOut = true,
                    lockoutRemainingSeconds = LOCKOUT_DURATION_SECONDS,
                    error = "Too many attempts. Please wait.",
                    isShakeError = true,
                )
            }
            startLockoutCountdown()
        } else {
            val remainingAttempts = MAX_ATTEMPTS - newFailedAttempts
            _uiState.update {
                it.copy(
                    isVerifying = false,
                    pin = "",
                    failedAttempts = newFailedAttempts,
                    error = "Incorrect PIN. $remainingAttempts attempt${if (remainingAttempts > 1) "s" else ""} remaining.",
                    isShakeError = true,
                )
            }
        }
    }

    private fun startLockoutCountdown() {
        lockoutJob?.cancel()
        lockoutJob = viewModelScope.launch {
            for (remaining in LOCKOUT_DURATION_SECONDS downTo 1) {
                _uiState.update { it.copy(lockoutRemainingSeconds = remaining) }
                delay(1000L)
            }
            // Lockout expired — reset
            _uiState.update {
                it.copy(
                    isLockedOut = false,
                    lockoutRemainingSeconds = 0,
                    failedAttempts = 0,
                    error = null,
                    isShakeError = false,
                )
            }
        }
    }

    /**
     * Called when the user taps "Use Biometric" button to re-trigger the prompt.
     */
    fun onBiometricRequested() {
        if (_uiState.value.isLockedOut) return
        _biometricEvent.tryEmit(BiometricEvent.ShowBiometricPrompt)
    }

    /**
     * Called when biometric authentication succeeds.
     */
    fun onBiometricSuccess() {
        _uiState.update {
            it.copy(failedAttempts = 0, error = null)
        }
        _navigationEvent.tryEmit(NavigationEvent.NavigateToMain)
    }

    /**
     * Called when biometric authentication fails or is cancelled.
     * User can still use PIN as fallback.
     */
    fun onBiometricError(errorMessage: String?) {
        // Don't show error for user cancellation — just let them use PIN
        // Only show error for actual failures
        if (errorMessage != null && !errorMessage.contains("cancel", ignoreCase = true)) {
            _uiState.update {
                it.copy(error = errorMessage)
            }
        }
    }

    /**
     * Clear shake error state after animation completes.
     */
    fun onShakeAnimationComplete() {
        _uiState.update { it.copy(isShakeError = false) }
    }

    override fun onCleared() {
        super.onCleared()
        lockoutJob?.cancel()
    }
}

**CRITICAL INSTRUCTIONS FOR THE AI:**
1. **PinInputField API** — same issue as Prompt 8. Open the actual `PinInputField` in
   `core/core-ui/` and check its parameter signature. Adapt accordingly. If PinInputField
   manages its own internal state and calls `onPinComplete(pin: String)`, then the
   ViewModel needs a different structure — have `onPinComplete` call `verifyPin` directly.
   Use `key(uiState.failedAttempts)` to force PinInputField recreation after each failed
   attempt (this resets its internal state).

2. **BiometricHelper integration** — this is the trickiest part. The BiometricHelper
   from Prompt 3 likely requires a `FragmentActivity` to show the biometric prompt.
   In Compose, get this via `LocalContext.current as? FragmentActivity`. The biometric
   prompt MUST be triggered from the composable layer (not the ViewModel) because it
   needs an Activity reference. The ViewModel emits `BiometricEvent.ShowBiometricPrompt`,
   and the `LaunchedEffect` in UnlockScreen handles it by calling `biometricHelper.authenticate()`.

3. **BiometricHelper injection** — BiometricHelper might be `@Inject`-able via Hilt.
   Options for providing it to UnlockScreen:
   - Pass it as a parameter from the NavHost/Activity level
   - Inject it into the ViewModel and expose a method (but it needs Activity, so this is awkward)

### 10. Common Issues to Watch For:

**Issue: BiometricHelper needs Activity but ViewModel can't hold Activity reference**
Solution: The ViewModel emits a `BiometricEvent`, and the Composable handles it in a
`LaunchedEffect` block where it has access to `LocalContext` → `FragmentActivity`.
This is the standard pattern for bridging Activity-dependent APIs with Compose+ViewModel.

**Issue: PinInputField resets after failed attempt**
If PinInputField manages internal state, use Compose's `key()` to force recreation:
```kotlin
key(uiState.failedAttempts) {
    PinInputField(...)
}
```
This creates a new PinInputField instance each time failedAttempts changes,
resetting any internal pin state.

**Issue: NavHost startDestination is static**
The `key(routingKey)` approach recreates the entire NavHost when the routing context
changes. This works but loses all navigation state. Since transitions between onboarding,
auth, and main are one-directional and complete, this is acceptable.

An alternative: use a single NavHost with all three graphs registered, start at a
"router" composable that navigates imperatively:
```kotlin
composable("router") {
    LaunchedEffect(isWalletCreated, isAuthed) {
        val destination = when {
            !isWalletCreated -> OnboardingRoutes.ONBOARDING_GRAPH
            !isAuthed -> AuthRoutes.AUTH_GRAPH
            else -> "main_graph"
        }
        navController.navigate(destination) {
            popUpTo("router") { inclusive = true }
        }
    }
}
```
Use whichever compiles and works correctly.

**Issue: Flash of wrong screen on launch**
When DataStore hasn't loaded yet, `isWalletCreated` defaults to `false`, causing
onboarding to show briefly. Fix options:
- Use `null` as initial value and show a loading/splash screen while null
- Use a `SplashScreen` API on the Activity level to hold until data loads
- Accept the brief flash as a known limitation (polish later)

**Issue: Auto-lock with runBlocking**
If `runBlocking` in `onStart` causes a lint warning or ANR concern, use the cached
value approach described in section 4.

**Issue: setWalletCreated not called anywhere**
If after checking the entire codebase, `preferences.setWalletCreated(true)` is never
called, add it to `SetPinViewModel.storePin()` (both create and import flows end there).
Also ensure `WalletRepositoryImpl.deleteWallet()` calls `preferences.setWalletCreated(false)`.
