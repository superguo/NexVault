/**
 * ViewModel for the Import Wallet screen.
 *
 * Supports two import modes:
 * 1. Mnemonic phrase (12 or 24 words separated by spaces)
 * 2. Raw private key (64 hex characters, optionally prefixed with 0x)
 *
 * Validates input in real time and shows appropriate error messages.
 * On successful import, navigates to the Set PIN screen.
 *
 * IMPORTANT: Adjust ImportWalletUseCase invocation to match actual API.
 * The use case might accept a sealed class like ImportInput.Mnemonic / ImportInput.PrivateKey,

@HiltViewModel
class ImportWalletViewModel @Inject constructor(
    private val importWalletUseCase: ImportWalletUseCase,
    // ^ Adjust type to match actual class name
) : ViewModel() {

    enum class ImportMode {
        MNEMONIC,
        PRIVATE_KEY,
    }

    data class UiState(
        val importMode: ImportMode = ImportMode.MNEMONIC,
        val mnemonicInput: String = "",
        val privateKeyInput: String = "",
        val mnemonicError: String? = null,
        val privateKeyError: String? = null,
        val isLoading: Boolean = false,
        val generalError: String? = null,
        val isImportEnabled: Boolean = false,
    )

    sealed interface NavigationEvent {
        data object NavigateToSetPin : NavigationEvent
    }

    private val _uiState = MutableStateFlow(UiState())
    val uiState: StateFlow<UiState> = _uiState.asStateFlow()

    private val _navigationEvent = MutableSharedFlow<NavigationEvent>(extraBufferCapacity = 1)
    val navigationEvent: SharedFlow<NavigationEvent> = _navigationEvent.asSharedFlow()

    fun onImportModeChanged(mode: ImportMode) {
        _uiState.update {
            it.copy(
                importMode = mode,
                mnemonicError = null,
                privateKeyError = null,
                generalError = null,
                isImportEnabled = when (mode) {
                    ImportMode.MNEMONIC -> isValidMnemonicFormat(it.mnemonicInput)
                    ImportMode.PRIVATE_KEY -> isValidPrivateKeyFormat(it.privateKeyInput)
                },
            )
        }
    }

    fun onMnemonicInputChanged(input: String) {
        val trimmed = input.lowercase().trim()
        val error = validateMnemonicInput(trimmed)
        _uiState.update {
            it.copy(
                mnemonicInput = input,
                mnemonicError = error,
                generalError = null,
                isImportEnabled = error == null && isValidMnemonicFormat(trimmed),
            )
        }
    }

    fun onPrivateKeyInputChanged(input: String) {
        val trimmed = input.trim()
        val error = validatePrivateKeyInput(trimmed)
        _uiState.update {
            it.copy(
                privateKeyInput = input,
                privateKeyError = error,
                generalError = null,
                isImportEnabled = error == null && isValidPrivateKeyFormat(trimmed),
            )
        }
    }

    fun onImportClicked() {
        val currentState = _uiState.value
        if (currentState.isLoading) return

        viewModelScope.launch {
            _uiState.update { it.copy(isLoading = true, generalError = null) }

            val result = when (currentState.importMode) {
                ImportMode.MNEMONIC -> {
                    val mnemonic = currentState.mnemonicInput.trim().lowercase()
                        .replace("\\s+".toRegex(), " ") // normalize whitespace
                    importWalletUseCase(mnemonic)
                    // ^ Adjust: might be importWalletUseCase.invoke(mnemonic)
                    // or importWalletUseCase.fromMnemonic(mnemonic)
                    // or importWalletUseCase(ImportInput.Mnemonic(mnemonic))
                }
                ImportMode.PRIVATE_KEY -> {
                    val key = currentState.privateKeyInput.trim()
                        .removePrefix("0x").removePrefix("0X")
                    importWalletUseCase(key)
                    // ^ Adjust: might be importWalletUseCase.fromPrivateKey(key)
                    // or importWalletUseCase(ImportInput.PrivateKey(key))
                }
            }

            when (result) {
                is DataResult.Success -> {
                    _uiState.update { it.copy(isLoading = false) }
                    _navigationEvent.tryEmit(NavigationEvent.NavigateToSetPin)
                }
                is DataResult.Error -> {
                    _uiState.update {
                        it.copy(
                            isLoading = false,
                            generalError = result.message
                                ?: "Import failed. Please check your input.",
                            // ^ Adjust field name for error message
                        )
                    }
                }
                else -> {
                    _uiState.update { it.copy(isLoading = false) }
                }
            }
        }
    }

    /**
     * Real-time validation for mnemonic input.
     * Returns error message or null if input is valid so far.
     * Note: this is format validation, not BIP-39 checksum validation.
     * Full checksum validation happens when the user taps Import.
     */
    private fun validateMnemonicInput(input: String): String? {
        if (input.isEmpty()) return null // no error on empty — just disable button

        val words = input.split("\\s+".toRegex()).filter { it.isNotEmpty() }
        if (words.size > 24) {
            return "Too many words. Enter 12 or 24 words."
        }

        // Check for non-alphabetic characters
        val hasInvalidChars = words.any { word -> !word.all { it.isLetter() } }
        if (hasInvalidChars) {
            return "Words should contain only letters."
        }

        // If word count is complete but not 12 or 24
        if (words.size in 13..23 || (words.size > 1 && words.size < 12 && !input.endsWith(" "))) {
            // Only show error if input looks "complete" (no trailing space = user stopped typing)
            // This is a heuristic — only show word count error when count is clearly wrong
            if (words.size !in listOf(12, 24) && !input.endsWith(" ") && words.size >= 12) {
                return "Enter exactly 12 or 24 words. Currently: ${words.size} words."
            }
        }

        return null
    }

    private fun isValidMnemonicFormat(input: String): Boolean {
        val words = input.split("\\s+".toRegex()).filter { it.isNotEmpty() }
        return words.size == 12 || words.size == 24
    }

    private fun validatePrivateKeyInput(input: String): String? {
        if (input.isEmpty()) return null

        val cleaned = input.removePrefix("0x").removePrefix("0X")

        if (cleaned.length > 64) {
            return "Private key is too long."
        }

        val hasInvalidChars = !cleaned.all { it in '0'..'9' || it in 'a'..'f' || it in 'A'..'F' }
        if (hasInvalidChars && cleaned.isNotEmpty()) {
            return "Private key must be hexadecimal (0-9, a-f)."
        }

        if (cleaned.isNotEmpty() && cleaned.length < 64) {
            // Don't show error while typing, but disable button
            return null
        }

        return null
    }

    private fun isValidPrivateKeyFormat(input: String): Boolean {
        val cleaned = input.removePrefix("0x").removePrefix("0X")
        return cleaned.length == 64 && cleaned.all { it in '0'..'9' || it in 'a'..'f' || it in 'A'..'F' }
    }
}

/**
 * ViewModel for the Set PIN screen.
 *
 * Two-phase flow:
 * 1. "Set Your PIN" — user enters a 6-digit PIN
 * 2. "Confirm Your PIN" — user re-enters the same PIN
 *
 * If PINs match, the PIN is stored and biometric preference is set.
 * If PINs don't match, error is shown and user resets to phase 1.
 *
 * Also handles optional biometric enable toggle.
 */
@HiltViewModel
class SetPinViewModel @Inject constructor(
    private val setPinUseCase: SetPinUseCase,
    private val authRepository: AuthRepository,
    // ^ Adjust types to match actual class names
) : ViewModel() {

    enum class PinPhase {
        SET,       // "Set Your PIN"
        CONFIRM,   // "Confirm Your PIN"
    }

    data class UiState(
        val phase: PinPhase = PinPhase.SET,
        val pin: String = "",
        val error: String? = null,
        val isLoading: Boolean = false,
        val isBiometricAvailable: Boolean = false,
        val isBiometricEnabled: Boolean = false,
        val isShakeError: Boolean = false,  // trigger shake animation on mismatch
    )

    sealed interface NavigationEvent {
        data object NavigateToMain : NavigationEvent
    }

    private val _uiState = MutableStateFlow(UiState())
    val uiState: StateFlow<UiState> = _uiState.asStateFlow()

    private val _navigationEvent = MutableSharedFlow<NavigationEvent>(extraBufferCapacity = 1)
    val navigationEvent: SharedFlow<NavigationEvent> = _navigationEvent.asSharedFlow()

    // Store the first PIN entry for comparison during confirm phase
    private var firstPin: String = ""

    init {
        checkBiometricAvailability()
    }

    private fun checkBiometricAvailability() {
        viewModelScope.launch {
            val available = authRepository.isBiometricAvailable()
            // ^ Adjust method name to match actual AuthRepository API
            _uiState.update { it.copy(isBiometricAvailable = available) }
        }
    }

    /**
     * Called when a digit is pressed on the PIN keypad.
     * Appends the digit to the current PIN.
     * When PIN reaches 6 digits, automatically handles phase transition.
     */
    fun onDigitPressed(digit: Int) {
        val currentState = _uiState.value
        if (currentState.isLoading || currentState.pin.length >= 6) return

        val newPin = currentState.pin + digit.toString()
        _uiState.update { it.copy(pin = newPin, error = null, isShakeError = false) }

        if (newPin.length == 6) {
            handlePinComplete(newPin)
        }
    }

    /**
     * Called when the backspace/delete button is pressed on the PIN keypad.
     * Removes the last digit from the current PIN.
     */
    fun onBackspacePressed() {
        val currentState = _uiState.value
        if (currentState.isLoading || currentState.pin.isEmpty()) return

        _uiState.update {
            it.copy(
                pin = it.pin.dropLast(1),
                error = null,
                isShakeError = false,
            )
        }
    }

    private fun handlePinComplete(pin: String) {
        when (_uiState.value.phase) {
            PinPhase.SET -> {
                // Store first PIN and move to confirm phase
                firstPin = pin
                _uiState.update {
                    it.copy(
                        phase = PinPhase.CONFIRM,
                        pin = "", // clear for second entry
                        error = null,
                    )
                }
            }
            PinPhase.CONFIRM -> {
                if (pin == firstPin) {
                    // PINs match — store and navigate
                    storePin(pin)
                } else {
                    // PINs don't match — show error and reset
                    _uiState.update {
                        it.copy(
                            isShakeError = true,
                            error = "PINs don't match. Please try again.",
                        )
                    }
                    // Reset to SET phase after a brief delay for the shake animation
                    viewModelScope.launch {
                        kotlinx.coroutines.delay(1000)
                        firstPin = ""
                        _uiState.update {
                            it.copy(
                                phase = PinPhase.SET,
                                pin = "",
                                error = null,
                                isShakeError = false,
                            )
                        }
                    }
                }
            }
        }
    }

    private fun storePin(pin: String) {
        viewModelScope.launch {
            _uiState.update { it.copy(isLoading = true) }

            try {
                // Store the PIN
                setPinUseCase(pin)
                // ^ Adjust: might be setPinUseCase.invoke(pin) or setPinUseCase.execute(pin)

                // Store biometric preference
                if (_uiState.value.isBiometricEnabled) {
                    authRepository.setBiometricEnabled(true)
                    // ^ Adjust method name. If this method doesn't exist on AuthRepository,
                    // it might be on NexVaultPreferences. Check and adapt.
                }

                _uiState.update { it.copy(isLoading = false) }
                _navigationEvent.tryEmit(NavigationEvent.NavigateToMain)
            } catch (e: Exception) {
                _uiState.update {
                    it.copy(
                        isLoading = false,
                        error = "Failed to set PIN: ${e.message}",
                    )
                }
            }
        }
    }

    fun onBiometricToggled(enabled: Boolean) {
        _uiState.update { it.copy(isBiometricEnabled = enabled) }
    }

    /**
     * Clear shake error state after animation completes.
     * Called from the composable after the shake animation finishes.
     */
    fun onShakeAnimationComplete() {
        _uiState.update { it.copy(isShakeError = false) }
    }
}

### 7. Common Issues to Watch For:

**Issue: PinInputField API mismatch**
If PinInputField is a fully self-contained component that manages its own pin state,
you need a way to reset it when the phase changes or an error occurs. Options:
(a) Add a `key` parameter and change it on reset: `PinInputField(key = phaseKey, ...)`.
    Using Compose's `key()` composable: `key(uiState.phase) { PinInputField(...) }`
(b) Add a `pin: String` external state parameter to PinInputField.
(c) Add a `reset: Boolean` trigger parameter.
Choose the simplest option that works with the existing PinInputField code.

**Issue: SingleChoiceSegmentedButtonRow not available**
If your Compose Material3 BOM doesn't include segmented buttons (requires BOM 2024.01.00+
or material3 1.2.0+), use this fallback:
```kotlin
@Composable
private fun ImportModeSelector(...) {
    TabRow(selectedTabIndex = if (selectedMode == ImportMode.MNEMONIC) 0 else 1) {
        Tab(selected = selectedMode == ImportMode.MNEMONIC,
            onClick = { onModeChanged(ImportMode.MNEMONIC) },
            text = { Text("Recovery Phrase") })
        Tab(selected = selectedMode == ImportMode.PRIVATE_KEY,
            onClick = { onModeChanged(ImportMode.PRIVATE_KEY) },
            text = { Text("Private Key") })
    }
}
```

**Issue: ImportWalletUseCase — mnemonic vs private key distinction**
The use case might expect different invocation patterns for each type. If it's a single
`invoke(input: String)` that auto-detects the type, that works too — just pass the
cleaned input. If it has separate methods like `fromMnemonic(mnemonic)` and
`fromPrivateKey(key)`, call the appropriate one based on `importMode`.

**Issue: Biometric preference storage location**
`setBiometricEnabled` might live on `NexVaultPreferences` (in core-datastore) rather
than `AuthRepository`. Check both interfaces and use whichever actually has this method.
If it's on NexVaultPreferences, inject that into the ViewModel as well.

**Issue: Loading overlay blocking interactions**
The loading overlay in SetPinScreen uses a semi-transparent Surface covering the
entire screen. This correctly blocks all touch interactions during PIN storage.
Make sure the scrim color is from the theme (`MaterialTheme.colorScheme.scrim`).

**Issue: Phase transition timing**
When transitioning from SET to CONFIRM phase, the PIN is cleared. If PinInputField
manages its own state, use `key(uiState.phase)` to force recomposition and reset:
```kotlin
key(uiState.phase) {
    PinInputField(...)
}
```
This creates a new PinInputField instance when the phase changes, clearing any
internal state.
